171 int ADC_Collect(unsigned int channelNum, unsigned int fs,
172  unsigned short buffer[], unsigned int numberOfSamples)
173 {
174  if(fs < 100 || fs > 10000){
175  return -1;
176  }
177  if(numberOfSamples < 1){
178  return -1;
179  }
180 
181  DisableInterrupts();
182  ADC0_ACTSS_R &= ~0x04; // disable sample sequencer 2
183  ADC0_EMUX_R = (ADC0_EMUX_R&0xFFFFF0FF)+0x0500; // timer trigger event SS2
184 
185  ADC0_SAC_R = 0x02; // 4x Hardware Oversample
186 
187  ADC0_SSMUX2_R = (channelNum << 4) | channelNum;
188  ADC0_SSCTL2_R = 0x064; // set flag and end, 2 samples at a time
189  ADC0_IM_R |= 0x04; // enable SS2 interrupts
190  ADC0_ACTSS_R |= 0x04; // enable sample sequencer 2
191  NVIC_PRI4_R = (NVIC_PRI4_R & 0xFFFF00FF)|0x00004000; //priority 2
192  NVIC_EN0_R = 1<<16; // enable interrupt 16 in NVIC
193  //Sample Sequencer 2
194  if(channelNum == 0){
195  Collecting[0] = 1;
196  SampleCount[0] = 0;
197  TargetCount[0] = numberOfSamples;
198  Buffer0 = buffer;
199  CurrentChannel = 0;
200  }
201  if(channelNum == 1){
202  Collecting[1] = 1;
203  SampleCount[1] = 0;
204  TargetCount[1] = numberOfSamples;
205  Buffer1 = buffer;
206  CurrentChannel = 1;
207  }
208  if(channelNum == 2){
209  Collecting[2] = 1;
210  SampleCount[2] = 0;
211  TargetCount[2] = numberOfSamples;
212  Buffer2 = buffer;
213  CurrentChannel = 2;
214  }
215  if(channelNum == 4){
216  Collecting[3] = 1;
217  SampleCount[3] = 0;
218  TargetCount[3] = numberOfSamples;
219  Buffer3 = buffer;
220  CurrentChannel = 3;
221  }
222  EnableInterrupts();
223 
224  return Collecting[0];
225 }
